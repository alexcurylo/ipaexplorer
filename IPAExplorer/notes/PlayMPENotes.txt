NSConditionLock
- (void)produceData
{
   [condition lock];
   // Produce new data
   newDataExists = YES;
   [condition signal];
   [condition unlock];
}
- (void)consumeData
{
   [condition lock];
   while (!newDataExists)
      [condition wait];

   // consume the new data
   newDataExists = NO;
   [condition unlock];
}
